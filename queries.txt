CREATE TABLE datacsv (
  clientIpAddress STRING COMMENT 'IP Address of the User',
  domainName STRING,
  remoteUser STRING,
  monthDate STRING,
  request STRING,
  httpStatusCode INT,
  bytesSent INT,
  referer STRING,
  userAgent STRING)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY '\;'
STORED AS TEXTFILE

LOAD DATA INPATH 'hdfs:///user/root/data.csv' INTO TABLE datacsv

CREATE TABLE datacsvpartitioned (
  clientIpAddress STRING COMMENT 'IP Address of the User',
  domainName STRING,
  remoteUser STRING,
  monthDate STRING,
  request STRING,
  httpStatusCode INT,
  bytesSent INT,
  referer STRING,
  userAgent STRING)
PARTITIONED BY (dn STRING)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY '\;'
STORED AS TEXTFILE
