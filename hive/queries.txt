#clientIpAddress;domainName;remoteUser;monthDate;request;httpStatusCode;bytesSent;referer;userAgent

CREATE EXTERNAL TABLE data (
  clientIpAddress STRING COMMENT 'IP Address of the User',
  domainName STRING,
  remoteUser STRING,
  monthDate STRING,
  request STRING,
  httpStatusCode INT,
  bytesSent INT,
  referer STRING,
  userAgent STRING)
ROW FORMAT DELIMITED
FIELDS TERMINATED BY '\;'
STORED AS TEXTFILE
location '/user/root/prepared/big';

CREATE TABLE dataparquet STORED AS parquet AS
SELECT * FROM data

SELECT monthDate, count(clientIpAddress) as total FROM
  (SELECT DISTINCT monthDate, clientIpAddress
   FROM `data`
   WHERE domainName = 'www.mydemo.com') mysubqueryresult
  GROUP BY monthDate
  ORDER BY monthDate ASC
LIMIT 10;
