SELECT monthDate, count(clientIpAddress) as total FROM
  (SELECT DISTINCT monthDate, clientIpAddress
   FROM `data`
   WHERE domainName = 'www.mydemo.com') mysubqueryresult GROUP BY monthDate ORDER BY monthDate ASC
LIMIT 10;
